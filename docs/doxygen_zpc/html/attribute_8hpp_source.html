130  template<typename T>
131  inline T get(attribute_store_node_value_state_t value_type) const
132  {
133  uint8_t buffer[ATTRIBUTE_STORE_MAXIMUM_VALUE_LENGTH];
134  uint8_t size = 0;
135  if (get(value_type, (uint8_t *)buffer, &size)) {
136  // For iterable containers we copy the data
137  if constexpr (is_iterable<T>::value) {
138  typedef typename T::value_type v_t;
139  //Make sure that value type of the conrainter int,float,bool ... etc
140  static_assert(std::is_trivially_copyable<v_t>::value);
141  T v;
142  v_t *start = reinterpret_cast<v_t *>(buffer);
143  v_t *end = reinterpret_cast<v_t *>(buffer + size);
144  std::copy(start, end, std::back_insert_iterator(v));
145  return v;
146  } else if constexpr (std::is_integral<T>::value) {
147  if (size == 1) {
148  return *(reinterpret_cast<uint8_t *>(buffer));
149  } else if (size == 2) {
150  return *(reinterpret_cast<uint16_t *>(buffer));
151  }
152  } else {
153  static_assert(std::is_trivially_copyable<T>::value);
154  }
155 
156  if (size == sizeof(T)) {
157  return *(reinterpret_cast<T *>(buffer));
158  } else {
159  throw std::invalid_argument(
160  "Unexpected attribute size for node " + std::to_string(_n)
161  + ".Expected: " + std::to_string(sizeof(T))
162  + ", actual: " + std::to_string(size)
163  + " value state: " + std::to_string(value_type) + ".");
164  }
165  } else {
166  throw std::invalid_argument("Unable to read attribute "
167  + std::to_string(_n));
168  }
169  }

475  int depth = 0)
476  {
477  sl_status_t status = func(*this, depth);
478  if (SL_STATUS_OK == status) {
