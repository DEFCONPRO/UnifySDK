350  * status if it does not receive the indication within a time period after
351  * receiving the ADD_NODE_STATUS_NODE_FOUND status.
352  * The RECOMMENDED timeout interval is 60 sec. */
353 #define ADD_NODE_STATUS_ADDING_END_NODE_TIMEOUT 60000
354 
355 /*
356  * INS13954 indicates that the timeout should be as follows:
357  * 1. for including an End Node:
358  * timeout = 76000 ms + listening nodes * 217 ms + FLiRS Nodes * 3517 ms
359  * 2. for including a Controller Node:
360  * timeout = 76000 ms + listening nodes * 217 ms + FLiRS Nodes * 3517 ms + network nodes * 732 ms
361  *
362  * This is the worst case scenario time out, we do not bother calculating the right timeout
363  */
364 #define ADD_NODE_PROTOCOL_NEIGHBOR_DISCOVERY_TIMEOUT \
365  (76000 + ZW_MAX_NODES * (3517 + 732))
