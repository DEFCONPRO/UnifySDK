385 /* An application MUST time out waiting for the ADD_NODE_STATUS_ADDING_END_NODE
386  * status if it does not receive the indication within a time period after
387  * receiving the ADD_NODE_STATUS_NODE_FOUND status.
388  * The RECOMMENDED timeout interval is 60 sec. */
389 #define ADD_NODE_STATUS_ADDING_END_NODE_TIMEOUT 60000
390 
391 /*
392  * INS13954 indicates that the timeout should be as follows:
393  * 1. for including an End Node:
394  * timeout = 76000 ms + listening nodes * 217 ms + FLiRS Nodes * 3517 ms
395  * 2. for including a Controller Node:
396  * timeout = 76000 ms + listening nodes * 217 ms + FLiRS Nodes * 3517 ms + network nodes * 732 ms
397  *
398  * This is the worst case scenario time out, we do not bother calculating the right timeout
399  */
400 #define ADD_NODE_PROTOCOL_NEIGHBOR_DISCOVERY_TIMEOUT \
401  (76000 + ZW_MAX_NODES * (3517 + 732))
