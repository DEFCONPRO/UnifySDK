417 #define LEARN_MODE_TIMEOUT 30000
418 
421 #define SEND_DATA_TIMEOUT 125000

423 /* An application MUST time out waiting for the ADD_NODE_STATUS_ADDING_END_NODE
424  * status if it does not receive the indication within a time period after
425  * receiving the ADD_NODE_STATUS_NODE_FOUND status.
426  * The RECOMMENDED timeout interval is 60 sec. */
427 #define ADD_NODE_STATUS_ADDING_END_NODE_TIMEOUT 60000
428 
429 /*
430  * INS13954 indicates that the timeout should be as follows:
431  * 1. for including an End Node:
432  * timeout = 76000 ms + listening nodes * 217 ms + FLiRS Nodes * 3517 ms
433  * 2. for including a Controller Node:
434  * timeout = 76000 ms + listening nodes * 217 ms + FLiRS Nodes * 3517 ms + network nodes * 732 ms
435  *
436  * This is the worst case scenario time out, we do not bother calculating the right timeout
437  */
438 #define ADD_NODE_PROTOCOL_NEIGHBOR_DISCOVERY_TIMEOUT \
439  (76000 + ZW_MAX_NODES * (3517 + 732))
