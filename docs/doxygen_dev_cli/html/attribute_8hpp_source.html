113  template<typename T>
114  inline T get(attribute_store_node_value_state_t value_type) const
115  {
116  uint8_t buffer[ATTRIBUTE_STORE_MAXIMUM_VALUE_LENGTH];
117  uint8_t size = 0;
118  if (get(value_type, (uint8_t *)buffer, &size)) {
119  // For iterable containers we copy the data
120  if constexpr (is_iterable<T>::value) {
121  typedef typename T::value_type v_t;
122  //Make sure that value type of the conrainter int,float,bool ... etc
123  static_assert(std::is_trivially_copyable<v_t>::value);
124  T v;
125  v_t *start = reinterpret_cast<v_t *>(buffer);
126  v_t *end = reinterpret_cast<v_t *>(buffer + size);
127  std::copy(start, end, std::back_insert_iterator(v));
128  return v;
129  } else if constexpr (std::is_integral<T>::value) {
130  if (size == 1) {
131  return *(reinterpret_cast<uint8_t *>(buffer));
132  } else if (size == 2) {
133  return *(reinterpret_cast<uint16_t *>(buffer));
134  }
135  } else {
136  static_assert(std::is_trivially_copyable<T>::value);
137  }
138 
139  if (size == sizeof(T)) {
140  return *(reinterpret_cast<T *>(buffer));
141  } else {
142  throw std::invalid_argument(
143  "Unexpected attribute size for node " + std::to_string(_n)
144  + ".Expected: " + std::to_string(sizeof(T))
145  + ", actual: " + std::to_string(size)
146  + " value state: " + std::to_string(value_type) + ".");
147  }
148  } else {
149  throw std::invalid_argument("Unable to read attribute "
150  + std::to_string(_n));
151  }
152  }

182  template<typename T>
183  inline attribute &set(attribute_store_node_value_state_t value_type, T value)
184  {
185  if constexpr (is_iterable<T>::value) {
186  typedef typename T::value_type v_t;
187  //Make sure that value type of the conrainter int,float,bool ... etc
188  static_assert(std::is_trivially_copyable<v_t>::value);
189 
190  //The purpose of this is to convert the iterable container into a linear buffer
191  std::vector<v_t> v;
192  std::copy(value.begin(), value.end(), std::back_insert_iterator(v));
193  return set(value_type,
194  reinterpret_cast<uint8_t *>(v.data()),
195  v.size() * sizeof(v_t));
196  } else {
197  static_assert(std::is_trivially_copyable<T>::value);
198  return set(value_type, reinterpret_cast<uint8_t *>(&value), sizeof(T));
199  }
