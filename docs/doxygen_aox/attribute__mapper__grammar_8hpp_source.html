92  {

94  blockCommentRule
95  = qi::lit("/*") >> *(qi::char_ - qi::lit("*/")) >> qi::lit("*/");
96  spaceRule = qi::space;

127 template<typename IteratorType, typename Skipper> class UAMGrammar :
128  public qi::grammar<IteratorType, ast::ast_tree(), Skipper>
129 {
130  public:
131  UAMGrammar() : UAMGrammar::base_type(start_rule, "Rules")
132  {
133  using qi::alnum;
134  using qi::alpha;
135  using qi::char_;
136  using qi::eol;
137  using qi::hex;
138  using qi::lit;
139  using qi::string;
140  using qi::uint_;
141 
142  using qi::fail;
143  using qi::on_error;
144  using namespace qi::labels;
145  using boost::phoenix::construct;
146  using boost::phoenix::val;
147  using boost::spirit::qi::real_parser;
148  using boost::spirit::qi::strict_ureal_policies;
149  //This parser parses real numbers, real numbers must contain a dot
150  real_parser<float, strict_ureal_policies<float>> ureal;
151 
152  // Parser operators: https://www.boost.org/doc/libs/1_80_0/libs/spirit/doc/html/spirit/qi/reference/operator.html
153 
154  // definitions, example: def my_var 1234
155  // note that here we use a schematic action to build the defs parser.
156  // in this way when the parse later passes an identifier the defs parser
157  // will output the expression from the def_rule.
158  def_rule
159  = "def" > (identifier_rule
160  > expression_rule)[boost::phoenix::bind(defs.add, _1, _2)];
161 
162  // identifier, name of something
163  identifier_rule = (alpha | char_("_")) >> *(alnum | char_("_"));
164 
165  // elements of an attribute path
166  attribute_path_element_rule
167  = '^' | (operand_rule >> '[' >> operand_rule >> ']') | operand_rule;
168 
169  // the whole attribute path, example d'1245.6666[3].9999
170  attribute_rule = char_("dre") >> '\'' >> attribute_path_element_rule % '.';
171 
172  // if condition, example if (r'1244 == 0) 2 4
173  condition_rule = "if" > operand_rule > operand_rule > operand_rule;
174 
175  // any operand type, numbers, attributes conditions etc.
176  operand_rule = "undefined" | ("0x" > hex) | (ureal >> 'f') | uint_
177  | attribute_rule | condition_rule | defs
178  | ('(' > expression_rule > ')') | (char_('-') > operand_rule)
179  | (char_('+') > operand_rule) | built_in_function_rule;
180 
181  // a term is made of 1 or more operators combined by a strong binding operator like * / | &
182  term_rule = operand_rule >> *(term_rule_operators > operand_rule);
183 
184  // an experssion is made of 1 or more operators combined by a weak binding operator like + and -
185  expression_rule = (term_rule >> *(expr_rule_operators > term_rule));
186 
187  // Functions can be put in assignments, and they contain an expression themselves
188  // fn_at_least_one_value_defined(r'2,r'3,r'5)
189  // fn_min_value(r'2,d'3 or r'3,r'5)
190  built_in_function_rule = "fn_" >> identifier_rule >> "("
191  >> (expression_rule > *("," >> expression_rule))
192  >> ")";
193 
194  // an assignment, example r'1234 = r'6666
195  assignment_rule = attribute_rule >> "=" >> expression_rule;
196 
197  // Scope configuration is used assign configurations for a scope
198  scope_setting_name = string("chain_reaction") | string("clear_desired")
199  | string("create_attributes")
200  | string("common_parent_type");
201  scope_setting_value = ("0x" > hex) | uint_;
202  scope_configuration_rule
203  = scope_setting_name > "(" > -scope_setting_value > ")";
204 
205  // the scope rule is used to group assignments and assign them a priority.
206  scope_rule = "scope" > uint_ > *scope_configuration_rule > "{"
207  >> +assignment_rule >> "}";
208 
209  // This is the entry rule for the grammer, definitions must come first, then scopes
210  start_rule = *def_rule > *scope_rule;
211 
212  expression_rule.name("expression");
213  term_rule.name("term");
214  operand_rule.name("operand");
215  condition_rule.name("condition");
216  attribute_rule.name("attribute");
217  assignment_rule.name("assignment");
218  built_in_function_rule.name("function");
219  scope_setting_name.name("scope_setting_name");
220  scope_setting_value.name("scope_setting_value");
221  scope_configuration_rule.name("scope_configuration_rule");
222  scope_rule.name("scope");
223  def_rule.name("def");
224  identifier_rule.name("identifier");
225  attribute_path_element_rule.name("path element");
226 
227  // Error handler for the parser
228  // See https://www.boost.org/doc/libs/1_76_0/libs/spirit/doc/html/spirit/qi/tutorials/mini_xml___error_handling.html
229  on_error<fail>(
230  start_rule,
231  std::cout << val("Error! Expecting ") << _4 // what failed?
232  << val(" here: \"")
233  << construct<std::string>(_3,
234  _2) // iterators to error-pos, end
235  << val("\"") << std::endl);
236  }
237 
238  private:
239  defines_t defs;
240 
241  // Rules ( small parsers ) used by the main parser.
242  // note that the second template argument is the data type of the output of the
243  // rule. The tokens produced by the parser must be compatible destination struct
244  // Ie if a rule parses and integer followed by a string then the struct in the
245  // second argument must have a integer as first argument and a string as second argument.
246  // See documentation for details:
247  // https://www.boost.org/doc/libs/1_76_0/libs/spirit/doc/html/spirit/qi/tutorials/employee___parsing_into_structs.html
248  qi::rule<IteratorType, ast::expression(), Skipper> expression_rule;
249  qi::rule<IteratorType, ast::expression(), Skipper> term_rule;
250  qi::rule<IteratorType, ast::operand(), Skipper> operand_rule;
251  qi::rule<IteratorType, ast::condition(), Skipper> condition_rule;
252  qi::rule<IteratorType, ast::attribute(), Skipper> attribute_rule;
253  qi::rule<IteratorType, ast::attribute_path_element(), Skipper>
254  attribute_path_element_rule;
255  qi::rule<IteratorType, std::string()> identifier_rule;
256  qi::rule<IteratorType, ast::function_invokation(), Skipper>
257  built_in_function_rule;
258  qi::rule<IteratorType, ast::assignment(), Skipper> assignment_rule;
259  qi::rule<IteratorType, std::string()> scope_setting_name;
260  qi::rule<IteratorType, boost::optional<unsigned int>, Skipper>
261  scope_setting_value;
262  qi::rule<IteratorType, ast::scope_setting(), Skipper>
263  scope_configuration_rule;
264  qi::rule<IteratorType, ast::scope(), Skipper> scope_rule;
265  qi::rule<IteratorType, Skipper> def_rule;
266  qi::rule<IteratorType, ast::ast_tree(), Skipper> start_rule;
267 };
