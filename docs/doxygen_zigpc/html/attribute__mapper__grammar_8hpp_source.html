110 template<typename IteratorType, typename Skipper> class UAMGrammar :
111  public qi::grammar<IteratorType, ast::ast_tree(), Skipper>
112 {
113  public:
114  UAMGrammar() : UAMGrammar::base_type(start_rule, "Rules")
115  {
116  using qi::alnum;
117  using qi::alpha;
118  using qi::char_;
119  using qi::hex;
120  using qi::uint_;
121 
122  using qi::fail;
123  using qi::on_error;
124  using namespace qi::labels;
125  using boost::phoenix::construct;
126  using boost::phoenix::val;
127 
128  // definitions, example: def my_var 1234
129  // note that here we use a schematic action to build the defs parser.
130  // in this way when the parse later passes an identifier the defs parser
131  // will output the expression from the def_rule.
132  def_rule
133  = "def" > (identifier_rule
134  > expression_rule)[boost::phoenix::bind(defs.add, _1, _2)];
135 
136  // identifier, name of something
137  identifier_rule = (alpha | char_("_")) >> *(alnum | char_("_"));
138 
139  // elements of an attribute path
140  attribute_path_element_rule
141  = '^' | (operand_rule >> '[' > operand_rule > ']') | operand_rule;
142 
143  // the whole attribute path, example d'1245.6666[3].9999
144  attribute_rule = char_("dre") >> '\'' >> attribute_path_element_rule % '.';
145 
146  // if condition, example if (r'1244 == 0) 2 4
147  condition_rule = "if" > operand_rule > operand_rule > operand_rule;
148 
149  // any operand type, numbers, attributes conditions etc.
150  operand_rule = "undefined" | ("0x" > hex) | uint_ | attribute_rule
151  | condition_rule | defs | ('(' > expression_rule > ')')
152  | (char_('-') > operand_rule) | (char_('+') > operand_rule);
153 
154  // a term is made of 1 or more operators combined by a strong binding operator like * / | &
155  term_rule = operand_rule >> *(term_rule_operators > operand_rule);
156 
157  // an experssion is made of 1 or more operators combined by a weak binding operator like + and -
158  expression_rule = (term_rule >> *(expr_rule_operators > term_rule));
159 
160  // an assignment, example r'1234 = r'6666
161  assignment_rule = attribute_rule >> "=" >> expression_rule;
162 
163  // the scope rule is used to group assignments and assign them a priority.
164  scope_rule = "scope" > uint_ >> "{" >> +assignment_rule >> "}";
165 
166  // This is the entry rule for the grammer, definitions must come first, then scopes
167  start_rule = *def_rule > *scope_rule;
168 
169  // Error handler for the parser
170  // See https://www.boost.org/doc/libs/1_76_0/libs/spirit/doc/html/spirit/qi/tutorials/mini_xml___error_handling.html
171  on_error<fail>(
172  start_rule,
173  std::cout << val("Error! Expecting ") << _4 // what failed?
174  << val(" here: \"")
175  << construct<std::string>(_3,
176  _2) // iterators to error-pos, end
177  << val("\"") << std::endl);
178  }
179 
180  private:
181  qi::symbols<char, ast::expression> defs;
182 
183  // Rules ( small parsers ) used by the main parser.
184  // note that the second template argument is the data type of the output of the
185  // rule. The tokens produced by the parser must be compatible destination struct
186  // Ie if a rule parses and integer followed by a string then the struct in the
187  // second argument must have a integer as first argument and a string as second argument.
188  // See documentation for details:
189  // https://www.boost.org/doc/libs/1_76_0/libs/spirit/doc/html/spirit/qi/tutorials/employee___parsing_into_structs.html
190  qi::rule<IteratorType, ast::expression(), Skipper> expression_rule;
191  qi::rule<IteratorType, ast::expression(), Skipper> term_rule;
192  qi::rule<IteratorType, ast::operand(), Skipper> operand_rule;
193  qi::rule<IteratorType, ast::condition(), Skipper> condition_rule;
194  qi::rule<IteratorType, ast::attribute(), Skipper> attribute_rule;
195  qi::rule<IteratorType, ast::attribute_path_element(), Skipper>
196  attribute_path_element_rule;
197  qi::rule<IteratorType, std::string()> identifier_rule;
198  qi::rule<IteratorType, ast::assignment(), Skipper> assignment_rule;
199  qi::rule<IteratorType, ast::scope(), Skipper> scope_rule;
200  qi::rule<IteratorType, Skipper> def_rule;
201  qi::rule<IteratorType, ast::ast_tree(), Skipper> start_rule;
202 };
