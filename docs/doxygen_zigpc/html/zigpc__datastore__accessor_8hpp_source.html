83  attribute_store_node_t find(attribute_store_node_t parent, search_key_t key)
84  {
85  auto found_node = ATTRIBUTE_STORE_INVALID_NODE;
86 
87  std::string key_str = std::to_string(key);
88 
89  bool input_valid = attribute_store_node_exists(parent);
90  if (input_valid == false) {
91  sl_log_warning(LOG_TAG,
92  "%s: %s entity[NodeId:%u] Failed to find parent node",
93  this->label.c_str(),
94  key_str.c_str());
95  } else {
96  found_node = attribute_store_get_node_child_by_value(
97  parent,
98  this->node_type,
99  DESIRED_ATTRIBUTE,
100  reinterpret_cast<uint8_t *>(&key),
101  sizeof(search_key_t),
102  0);
103  }
104 
105  return found_node;

116  size_t child_count(attribute_store_node_t parent,
117  search_key_t key,
118  attribute_store_type_t child_type)
119  {
120  size_t child_count = 0U;
121 
122  attribute_store_node_t found_node = this->find(parent, key);
123  if (found_node != ATTRIBUTE_STORE_INVALID_NODE) {
124  attribute_store_node_t child_node
125  = attribute_store_get_node_child_by_type(found_node,
126  child_type,
127  child_count);
128 
129  bool child_valid = !(child_node == ATTRIBUTE_STORE_INVALID_NODE);
130  while (child_valid) {
131  child_count++;
132  child_node = attribute_store_get_node_child_by_type(found_node,
133  child_type,
134  child_count);
135  child_valid = !(child_node == ATTRIBUTE_STORE_INVALID_NODE);
136  }
137  }
138 
139  return child_count;
140  }

185  sl_status_t create(attribute_store_node_t parent, search_key_t key)
186  {
187  sl_status_t status = SL_STATUS_OK;
188  auto found_node = ATTRIBUTE_STORE_INVALID_NODE;
189 
190  std::string key_str = std::to_string(key);
191 
192  found_node = this->find(parent, key);
193  if (found_node != ATTRIBUTE_STORE_INVALID_NODE) {
194  status = SL_STATUS_ALREADY_EXISTS;
195  } else {
196  found_node = attribute_store_add_node(this->node_type, parent);
197  if (found_node == ATTRIBUTE_STORE_INVALID_NODE) {
198  status = SL_STATUS_FAIL;
199  sl_log_error(LOG_TAG,
200  "%s: Failed to create entity %s",
201  this->label.c_str(),
202  key_str.c_str());
203  }
204  }
205 
206  if (status == SL_STATUS_OK) {
207  uint8_t key_size = sizeof(search_key_t);
208 
209  status = attribute_store_set_node_attribute_value(
210  found_node,
211  DESIRED_ATTRIBUTE,
212  reinterpret_cast<uint8_t *>(&key),
213  key_size);
214  if (status != SL_STATUS_OK) {
215  sl_log_error(LOG_TAG,
216  "%s: %s entity[NodeId:%u]: Failed to populate entity data",
217  this->label.c_str(),
218  key_str.c_str(),
219  found_node);
220  }
221  }
222 
223  return status;

254  status = attribute_store_get_node_attribute_value(

360  sl_status_t remove_children(attribute_store_node_t parent, search_key_t key)
361  {
362  sl_status_t status = SL_STATUS_OK;
363 
364  attribute_store_node_t found_node = this->find(parent, key);
365 
366  std::string key_str = std::to_string(key);
367  if (found_node == ATTRIBUTE_STORE_INVALID_NODE) {
368  status = SL_STATUS_NOT_FOUND;
369  sl_log_warning(LOG_TAG,
370  "%s: Failed to find entity %s to remove children under",
371  this->label.c_str(),
372  key_str.c_str());
373  } else {
374  size_t child_count = attribute_store_get_node_child_count(found_node);
375  for (size_t i = 0; i < child_count; i++) {
376  attribute_store_node_t child_node
377  = attribute_store_get_node_child(found_node, i);
378  if (child_node != ATTRIBUTE_STORE_INVALID_NODE) {
379  status = attribute_store_delete_node(child_node);
380  if (status != SL_STATUS_OK) {
381  sl_log_error(LOG_TAG,
382  "%s: Failed to remove child[%u] remove entity",
383  this->label.c_str(),
384  i);
385  }
386  }
387  }
388  }
389 
390  return status;
391  }

402  virtual std::string to_str(attribute_store_node_t parent, search_key_t key)
403  {
404  std::ostringstream ss;
405  data_t data;
406 
407  sl_status_t status = this->read(parent, key, &data);
408  if (status == SL_STATUS_OK) {
409  ss << this->label << ": ";
410  uint8_t *buffer = reinterpret_cast<uint8_t *>(&data);
411  for (size_t i = 0U; i < sizeof(data_t); i++) {
412  ss << *buffer << " ";
413  buffer++;
414  }
415  }
416 
417  return ss.str();
418  }

445  sl_status_t read_list(attribute_store_node_t list_parent,
446  list_entry_t *const list,
447  size_t list_capacity)
448  {
449  sl_status_t status = SL_STATUS_OK;
450 
451  if (list == nullptr) {

453  }

455  status = this->list_entry.find_by_index(list_parent, i, list + i);
456  }
457  return status;

475  sl_status_t create_list(attribute_store_node_t list_parent,
476  const list_entry_t *list,
477  size_t list_count)
478  {
479  sl_status_t status = SL_STATUS_OK;
480 
481  if (list == nullptr) {
482  status = SL_STATUS_NULL_POINTER;
483  } else {
484  // Ensure there is no existing array under the parent by checking if
485  // first element exists.
486  attribute_store_node_t child_node
487  = attribute_store_get_node_child_by_type(
488  list_parent,
489  this->list_entry.get_node_type(),
490  0);
491  if (child_node != ATTRIBUTE_STORE_INVALID_NODE) {
492  status = SL_STATUS_ALREADY_EXISTS;
493  }
494  }
495 
496  for (size_t i = 0; (status == SL_STATUS_OK) && (i < list_count); i++) {
497  status = this->list_entry.create(list_parent, list[i]);
498  }
499 
500  return status;
501  }
502 
509  sl_status_t remove_list(attribute_store_node_t list_parent)
510  {
511  sl_status_t status = SL_STATUS_OK;
512 
513  attribute_store_type_t entry_type = this->list_entry.get_node_type();
514 
515  if (list_parent != ATTRIBUTE_STORE_INVALID_NODE) {
516  size_t list_index = 0U;
517 
518  attribute_store_node_t entry_node
519  = attribute_store_get_node_child_by_type(list_parent,
520  entry_type,
521  list_index);
522  while ((status == SL_STATUS_OK)
523  && entry_node != ATTRIBUTE_STORE_INVALID_NODE) {
524  status = attribute_store_delete_node(entry_node);
525  list_index++;
526  entry_node = attribute_store_get_node_child_by_type(list_parent,
527  entry_type,
528  list_index);
529  }
530  }
531 
532  return status;
533  }
